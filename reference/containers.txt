
structs:  field - field

1d arrays:  length - firstchunk -> chunk -> chunk

	chunk:  next - 256 byte array

	* last chunk points to null

tables:  length - width - firstchunk -> chunk -> (chunk, chunk)

	chunk:  down - right - i - j - data

	* edge chunks point to null

lists:  node -> node

	node:  down - right

	* in branch nodes, right points to the next node and down points to a node lower in the tree
	* in leaf nodes, next1 points to null and next2 points to the actual data (or to null)
